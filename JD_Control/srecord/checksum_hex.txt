# srec_cat command file to dump the CRC for a code area
# Usage: srec_cat @filename
..\Debug\JD_Ctrl.srec          		# input file
-fill 0xFF 0x8000000 0x8080000     	# fill code area with 0xff
-crop 0x8000000 0x807FFFE    		# just keep code area for CRC calculation below (CRC will be at 0x1FFFE..0x1FFFF)
-CRC16_Little_Endian 0x807FFFE -CCITT  	# calculate big endian CCITT CRC16 at given address.
-crop 0x8000000 0x8080000		# keep the CRC itself
-Output  JD_Ctrl.hex -Intel # produce output
